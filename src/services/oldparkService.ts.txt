  async getParkById(id: string): Promise<IPark | undefined> {
    console.log(`getParkById called with id=${id}`);

    try {
      const response = await fetch("https://parksapi.547bikes.info/api/Parks");
      console.log("API responded with status:", response.status);

      if (response.ok) {
        const parks = await response.json();
        const p = parks.find((park: any) => park.id === id);

        if (p) {
          const park: IPark = {
            id: p.id,
            parkName: p.name,
            location: p.address,
            description: p.description,
            adultPrice: p.adultPrice,
            childPrice: p.childPrice,
            imageUrl: p.pic1url,
            reviews: p.reviews ?? [],
          };

          console.log("Returning matched park:", park);
          return park;
        }
      }
    } catch (err) {
      console.error("Error in getParkById:", err);
    }

    const fallback = mockData.find((p) => p.id === id);
    console.log("Returning park from mockData:", fallback);
    return fallback;
  }
}

// If you want getAllParksWithReviews as a standalone function:
export const getAllParks = async (): Promise<IPark[]> => {
  console.log("getAllParksWithReviews called...");

  try {
    const [parksResponse, reviewsResponse] = await Promise.all([
      fetch("https://parksapi.547bikes.info/api/Parks"),
      fetch("https://parksapi.547bikes.info/api/ParkReview"),
    ]);

    if (!parksResponse.ok || !reviewsResponse.ok) {
      throw new Error("Failed to fetch parks or reviews");
    }

    const parksJson = await parksResponse.json();
    const reviewsJson = await reviewsResponse.json();

    console.log("Raw Parks JSON:", parksJson);
    console.log("Raw Reviews JSON:", reviewsJson);

    const filteredParks = parksJson.filter((p: any) => p.motocross === 1);

    const mappedReviews: Review[] = reviewsJson.map((r: any) => ({
      author: {
        id: r.useridasstring ?? String(r.userId),
        displayName: r.useridasstring ?? "Unknown User",
        fullName: "Unknown",
        dateOfBirth: new Date(),
      },
      rating: r.stars,
      review: r.description,
      dateWritten: r.datePosted ? new Date(r.datePosted) : undefined,
      dateVisited: undefined,
      active: !!r.dateApproved && !r.dateDenied,
    }));

    const reviewsByPark: Record<number, Review[]> = {};
    mappedReviews.forEach((rev: any) => {
      const pid = rev.parkId;
      if (!reviewsByPark[pid]) reviewsByPark[pid] = [];
      reviewsByPark[pid].push(rev);
    });

    const apiParks: IPark[] = filteredParks.map((p: any) => ({
      id: p.id,
      parkName: p.name,
      location: p.address,
      description: p.description,
      adultPrice: p.adultPrice,
      childPrice: p.childPrice,
      imageUrl: p.pic1url,
      motocross: p.motocross,
      reviews: reviewsByPark[p.id] ?? [],
    }));

    const combined: IPark[] = [...apiParks, ...mockData];
    console.log("Combined Parks (API + mockData):", combined);

    return combined;
  } catch (err) {
    console.error("Error in getAllParksWithReviews:", err);
    return mockData;
  }
};
